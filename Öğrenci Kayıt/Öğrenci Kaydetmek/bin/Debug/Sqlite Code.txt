SELECT SUM (OrderQuantity) AS [Toplam Sipariþ Adeti], 
       (SELECT EnglishProductName 
               FROM DimProduct 
               WHERE ProductKey = FactResellerSales.ProductKey LIMIT 1) AS [Ürün Adý]
FROM   FactResellerSales
WHERE  OrderDate BETWEEN '2001-01-01' AND '2005-12-30' 
GROUP  BY ProductKey
ORDER  BY 1 DESC
LIMIT 5
